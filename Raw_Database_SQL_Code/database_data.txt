--Create table for warehouse 
CREATE TABLE WAREHOUSE (
    Warehouse_No INT NOT NULL,
    Address VARCHAR(100) NOT NULL,
    PRIMARY KEY (Warehouse_No)
);

--Create table for stores product
CREATE TABLE STORES_PRODUCT (
    Warehouse_No INT NOT NULL,
    Product_Number INT NOT NULL,
    Quantity INT NOT NULL DEFAULT 0,
    FOREIGN KEY (Warehouse_No) REFERENCES WAREHOUSE(Warehouse_No) ON DELETE CASCADE,
    FOREIGN KEY (Product_Number) REFERENCES PRODUCT(Product_Number) ON DELETE CASCADE,
    PRIMARY KEY (Warehouse_No, Product_Number)
);


--Create table for product
CREATE TABLE PRODUCT (
    Product_Number INT NOT NULL,
    Price DECIMAL(10, 2) NOT NULL DEFAULT 0.00,
    Product_Type VARCHAR(100) NOT NULL,
    PRIMARY KEY (Product_Number)

);

--Create table for drinks
CREATE TABLE DRINK (
    Product_Number INT NOT NULL,
    Drink_Name VARCHAR(100) NOT NULL,
    FOREIGN KEY (Product_Number) REFERENCES PRODUCT(Product_Number) ON DELETE CASCADE,
    PRIMARY KEY (Drink_Name)

);

--Create table for board games
CREATE TABLE BOARD_GAME (
    Product_Number INT NOT NULL,
    Game_Name VARCHAR(100) NOT NULL,
    FOREIGN KEY (Product_Number) REFERENCES PRODUCT(Product_Number) ON DELETE CASCADE,
    PRIMARY KEY (Game_Name)

);

--Create table for books
CREATE TABLE BOOK (
    ISBN INT NOT NULL,
    Product_Number INT NOT NULL,
    Book_Title VARCHAR(100) NOT NULL,
    Publisher_ID INT NOT NULL,
    FOREIGN KEY (Product_Number) REFERENCES PRODUCT(Product_Number) ON DELETE CASCADE,
    FOREIGN KEY (Publisher_ID) REFERENCES PUBLISHER(Publisher_ID) ON DELETE CASCADE,
    PRIMARY KEY (ISBN)
);

--Create table for publishers
CREATE TABLE PUBLISHER (
    Publisher_ID INT NOT NULL,
    Publisher_NAME VARCHAR(100) NOT NULL,
    PRIMARY KEY (Publisher_ID)
);

--Create table for authors
CREATE TABLE AUTHOR (
    Author_ID INT NOT NULL,
    First_Name VARCHAR(100) NOT NULL,
    Middle_Name VARCHAR(100),
    Last_Name VARCHAR(100) NOT NULL,
    PRIMARY KEY (Author_ID)
);

--Create table for employees
CREATE TABLE EMPLOYEE (
    Employee_ID INT NOT NULL,
    Store_ID INT NOT NULL,
    Sales_No INT NOT NULL,
    Shift VARCHAR(20) NOT NULL DEFAULT '9AM-5PM',
    First_Name VARCHAR(100) NOT NULL,
    Last_Name VARCHAR(100) NOT NULL,
    Dept_No INT NOT NULL, 
    FOREIGN KEY (Store_ID) REFERENCES BOOK_STORE(Store_ID) ON DELETE CASCADE,
    FOREIGN KEY(Dept_No) REFERENCES DEPARTMENT(Dept_No) ON DELETE CASCADE,
    PRIMARY KEY (Employee_ID)
);

--Create table for departments
CREATE TABLE DEPARTMENT (
    Dept_No INT NOT NULL,
    Store_ID INT NOT NULL,
    Dept_Name VARCHAR(100) NOT NULL,
    Dept_Address VARCHAR(100) NOT NULL,
    FOREIGN KEY (Store_ID) REFERENCES BOOK_STORE(Store_ID) ON DELETE CASCADE,
    PRIMARY KEY (Dept_No)
);

--Create table for managers in general
CREATE TABLE MANAGER (
    Employee_ID INT NOT NULL,
    Store_ID INT NOT NULL,
    Sales_No INT NOT NULL,
    Shift VARCHAR(20) NOT NULL DEFAULT '9AM-5PM',
    First_Name VARCHAR(100) NOT NULL,
    Last_Name VARCHAR(100) NOT NULL,
    Manager_ID INT NOT NULL,
    Manager_Type VARCHAR(100) NOT NULL,
    Dept_No INT NOT NULL,
    FOREIGN KEY (Employee_ID) REFERENCES EMPLOYEE(Employee_ID) ON DELETE CASCADE,
    FOREIGN KEY (Store_ID) REFERENCES BOOK_STORE(Store_ID) ON DELETE CASCADE,
    FOREIGN KEY (Manager_ID) REFERENCES EMPLOYEE(Employee_ID) ON DELETE CASCADE,
    FOREIGN KEY(Dept_No) REFERENCES DEPARTMENT(Dept_No) ON DELETE CASCADE,
    PRIMARY KEY (Employee_ID)
);

--Create table for general employees
CREATE TABLE GENERAL_EMPLOYEE (
    Employee_ID INT NOT NULL,
    Store_ID INT NOT NULL,
    Sales_No INT NOT NULL,
    Shift VARCHAR(20) NOT NULL DEFAULT '9AM-5PM',
    First_Name VARCHAR(100) NOT NULL,
    Last_Name VARCHAR(100) NOT NULL,
    Manager_ID INT NOT NULL,
    Dept_No INT NOT NULL,
    FOREIGN KEY (Employee_ID) REFERENCES EMPLOYEE(Employee_ID) ON DELETE CASCADE,
    FOREIGN KEY (Store_ID) REFERENCES BOOK_STORE(Store_ID) ON DELETE CASCADE,
    FOREIGN KEY(Dept_No) REFERENCES DEPARTMENT(Dept_No) ON DELETE CASCADE,
    PRIMARY KEY (Employee_ID)
);

--Create table for managers who manage employees
CREATE TABLE EMPLOYEE_MANAGER (
    Employee_ID INT NOT NULL,
    Store_ID INT NOT NULL,
    Sales_No INT NOT NULL,
    Shift VARCHAR(20) NOT NULL DEFAULT '9AM-5PM',
    First_Name VARCHAR(100) NOT NULL,
    Last_Name VARCHAR(100) NOT NULL,
    Manager_ID INT NOT NULL,
    Manager_Type VARCHAR(100) NOT NULL,
    Dept_No INT NOT NULL,
    FOREIGN KEY (Employee_ID) REFERENCES EMPLOYEE(Employee_ID) ON DELETE CASCADE,
    FOREIGN KEY (Store_ID) REFERENCES BOOK_STORE(Store_ID) ON DELETE CASCADE,
    FOREIGN KEY (Manager_ID) REFERENCES EMPLOYEE(Employee_ID) ON DELETE CASCADE,
    FOREIGN KEY(Dept_No) REFERENCES DEPARTMENT(Dept_No) ON DELETE CASCADE,
    PRIMARY KEY (Employee_ID)
);

--Create table for managers who manage stores
CREATE TABLE STORE_MANAGER (
    Employee_ID INT NOT NULL,
    Store_ID INT NOT NULL,
    Sales_No INT NOT NULL,
    Shift VARCHAR(20) NOT NULL DEFAULT '9AM-5PM',
    First_Name VARCHAR(100) NOT NULL,
    Last_Name VARCHAR(100) NOT NULL,
    Manager_ID INT NOT NULL,
    Manager_Type VARCHAR(100) NOT NULL,
    Dept_No INT NOT NULL,
    FOREIGN KEY (Employee_ID) REFERENCES EMPLOYEE(Employee_ID) ON DELETE CASCADE,
    FOREIGN KEY (Store_ID) REFERENCES BOOK_STORE(Store_ID) ON DELETE CASCADE,
    FOREIGN KEY (Manager_ID) REFERENCES EMPLOYEE(Employee_ID) ON DELETE CASCADE,
    PRIMARY KEY (Employee_ID)
);

--Create table for book stores
CREATE TABLE BOOK_STORE (
    Store_ID INT NOT NULL,
    Address VARCHAR(100) NOT NULL,
    PRIMARY KEY (Store_ID)
);

--Create table for purchases
CREATE TABLE PURCHASE (
    Purchase_ID INT NOT NULL,
    Shipping_Address VARCHAR(100) NOT NULL,
    Billing_Address VARCHAR(100) NOT NULL,
    Payment_Method VARCHAR(100) NOT NULL,
    Date DATE NOT NULL,
    PRIMARY KEY (Purchase_ID)
);

--Create table for customers
CREATE TABLE CUSTOMER (
    Loyalty_No INT NOT NULL,
    Email_Address VARCHAR(100) NOT NULL,
    Address VARCHAR(100) NOT NULL,
    First_Name VARCHAR(100) NOT NULL,
    Last_Name VARCHAR(100) NOT NULL,
    PRIMARY KEY (Loyalty_No)
);

--Create table for customers who make purchases
CREATE TABLE MAKES_PURCHASE (
    Loyalty_No INT NOT NULL,
    Purchase_ID INT NOT NULL,
    FOREIGN KEY (Loyalty_No) REFERENCES CUSTOMER(Loyalty_No) ON DELETE CASCADE,
    FOREIGN KEY (Purchase_ID) REFERENCES PURCHASE(Purchase_ID) ON DELETE CASCADE,
    PRIMARY KEY (Loyalty_No, Purchase_ID)
);

--Create table that holds which author writes which book
CREATE TABLE WRITTEN_BY (
    Author_ID INT NOT NULL,
    ISBN INT NOT NULL,
    FOREIGN KEY (Author_ID) REFERENCES AUTHOR(Author_ID) ON DELETE CASCADE,
    FOREIGN KEY (ISBN) REFERENCES BOOK(ISBN) ON DELETE CASCADE,
    PRIMARY KEY (Author_ID, ISBN)
);

--Create table that holds which employee stocks which products
CREATE TABLE STOCKED_BY (
    Employee_ID INT NOT NULL,
    Product_Number INT NOT NULL,
    FOREIGN KEY (Employee_ID) REFERENCES EMPLOYEE(Employee_ID) ON DELETE CASCADE,
    FOREIGN KEY (Product_Number) REFERENCES PRODUCT(Product_Number) ON DELETE CASCADE,
    PRIMARY KEY (Employee_ID, Product_Number)
);

--Create table that holds the relationship between products and purchases
CREATE TABLE PRODUCT_PURCHASE(
    Product_Number INT NOT NULL,
    Purchase_ID INT NOT NULL,
    Quantity INT NOT NULL,
    FOREIGN KEY(Product_Number) REFERENCES PRODUCT(Product_Number) ON DELETE CASCADE, 
    FOREIGN KEY(Purchase_ID) REFERENCES PURCHASE(Purchase_ID) ON DELETE CASCADE ,
    PRIMARY KEY(Product_Number, Purchase_ID)
);

--Generate views
--Create view for viewing which book a customer purchases and the transaction details for cash transactions
CREATE VIEW CASH_CUSTOMER_TRANSACTIONS AS
SELECT CUSTOMER.First_Name, CUSTOMER.Last_Name, CUSTOMER.Loyalty_No, CUSTOMER.Email_Address, PURCHASE.Purchase_ID,  PURCHASE.Billing_Address,PURCHASE.Shipping_Address , PURCHASE.Date, PURCHASE.Payment_Method, BOOK.Book_Title
FROM CUSTOMER, MAKES_PURCHASE, PURCHASE, BOOK, PRODUCT_PURCHASE
WHERE CUSTOMER.Loyalty_No= MAKES_PURCHASE.Loyalty_No AND MAKES_PURCHASE.Purchase_ID= PURCHASE.Purchase_ID AND PRODUCT_PURCHASE.Purchase_ID= PURCHASE.Purchase_ID AND PRODUCT_PURCHASE.Product_Number= BOOK.Product_Number AND PURCHASE.Payment_Method= 'CASH';

--Create view for looking at which employees work at a store managed by a certain store manager in this example the manager is John Smith
CREATE VIEW MANAGED_EMPLOYEE_INFO AS
SELECT S.Employee_ID As Store_Manager_ID, S.First_Name AS Store_Manager_First_Name, S.Last_Name AS Store_Manager_Last_Name, E.Employee_ID , E.First_Name , E.Last_Name, E.Sales_No AS Sales_Made_By_Employee,     (SELECT COUNT(*) FROM STOCKED_BY WHERE Employee_ID = E.Employee_ID) AS Products_Stocked
FROM GENERAL_EMPLOYEE E, STORE_MANAGER S
WHERE E.Store_ID= S.Store_ID AND S.First_Name= 'John' AND S.Last_Name= 'Smith'
ORDER BY S.Store_ID;

--Create view for viewing product details that are stored in various warehouses in this example it is first warehouse and focuses on Romance Books
CREATE VIEW PRODUCT_WAREHOUSE_STORAGE AS
SELECT DISTINCT W.Warehouse_No, W.Address, P.Product_Number,  P.Product_Type
FROM STORES_PRODUCT S, WAREHOUSE W, PRODUCT P
WHERE S.Product_Number= P.Product_Number AND W.Warehouse_No=S.Warehouse_No  AND P.Product_Number AND W.Warehouse_No=1 AND P.Product_Type= 'Romance Book';
